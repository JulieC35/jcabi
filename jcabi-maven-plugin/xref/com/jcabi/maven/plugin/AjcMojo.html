162     /**
163      * {@inheritDoc}
164      */
165     @Override
166     @Loggable(value = Loggable.DEBUG, limit = 1, unit = TimeUnit.MINUTES)
167     public void execute() throws MojoFailureException {
168         StaticLoggerBinder.getSingleton().setMavenLog(this.getLog());
169         this.classesDirectory.mkdirs();
170         this.tempDirectory.mkdirs();
171         final Main main = new Main();
172         final IMessageHolder mholder = new AjcMojo.MsgHolder();
173         final String jdk = "1.6";
174         main.run(
175             new String[] {
176                 "-Xset:avoidFinal=true",
177                 "-inpath",
178                 this.classesDirectory.getAbsolutePath(),
179                 "-sourceroots",
180                 this.sourceroots(),
181                 "-d",
182                 this.tempDirectory.getAbsolutePath(),
183                 "-classpath",
184                 StringUtils.join(this.classpath(), AjcMojo.SEP),
185                 "-aspectpath",
186                 this.aspectpath(),
187                 "-source",
188                 jdk,
189                 "-target",
190                 jdk,
191                 "-g:none",
192                 "-encoding",
193                 "UTF-8",
194                 "-time",
195                 "-showWeaveInfo",
196                 "-warn:constructorName",
197                 "-warn:packageDefaultMethod",
198                 "-warn:deprecation",
199                 "-warn:maskedCatchBlocks",
200                 "-warn:unusedLocals",
201                 "-warn:unusedArguments",
202                 "-warn:unusedImports",
203                 "-warn:syntheticAccess",
204                 "-warn:assertIdentifier",
205             },
206             mholder
207         );
208         try {
209             FileUtils.copyDirectory(this.tempDirectory, this.classesDirectory);
210         } catch (IOException ex) {
211             throw new MojoFailureException("failed to copy files back", ex);
212         }
213         Logger.info(
214             this,
215             // @checkstyle LineLength (1 line)
216             "ajc result: %d file(s) processed, %d pointcut(s) woven, %d error(s), %d warning(s)",
217             AjcMojo.files(this.tempDirectory).size(),
218             mholder.numMessages(IMessage.WEAVEINFO, false),
219             mholder.numMessages(IMessage.ERROR, true),
220             mholder.numMessages(IMessage.WARNING, false)
221         );
222         if (mholder.hasAnyMessage(IMessage.ERROR, true)) {
223             throw new MojoFailureException("AJC failed, see log above");
224         }
225     }

227     /**
228      * Get classpath for AJC.
229      * @return Classpath
230      */
231     @Cacheable(forever = true)
232     @Loggable(
233         value = Loggable.DEBUG,
234         limit = 1, unit = TimeUnit.MINUTES,
235         trim = false
236     )
237     private Collection<File> classpath() {
238         final Collection<String> scps;
239         if (this.scopes == null) {
240             scps = this.scope();
241         } else {
242             scps = Arrays.asList(this.scopes);
243         }
244         final Collection<File> files = new LinkedList<File>();
245         try {
246             final DependencyGraphBuilder builder =
247                 (DefaultDependencyGraphBuilder) this.container.lookup(
248                     DependencyGraphBuilder.class.getCanonicalName(),
249                     "default"
250                 );
251             final DependencyNode node = builder.buildDependencyGraph(
252                 this.project,
253                 new ArtifactFilter() {
254                     @Override
255                     public boolean include(final Artifact artifact) {
256                         return scps.contains(artifact.getScope());
257                     }
258                 }
259             );
260             files.addAll(this.dependencies(node, scps));
261         } catch (DependencyGraphBuilderException ex) {
262             throw new IllegalStateException(ex);
263         } catch (ComponentLookupException ex) {
264             throw new IllegalStateException(ex);
265         }
266         return files;
267     }

269     /**
270      * Retrieve dependencies for from given node and scope.
271      * @param node Node to traverse.
272      * @param scps Scopes to use.
273      * @return Collection of dependency files.
274      */
275     private Collection<File> dependencies(final DependencyNode node,
276         final Collection<String> scps) {
277         final Artifact artifact = node.getArtifact();
278         final Collection<File> files = new LinkedList<File>();
279         if ((artifact.getScope() == null)
280             || scps.contains(artifact.getScope())) {
281             if (artifact.getScope() == null) {
282                 files.add(artifact.getFile());
283             } else {
284                 files.add(
285                     this.session.getLocalRepository().find(artifact).getFile()
286                 );
287             }
288             for (DependencyNode child : node.getChildren()) {
289                 if (child.getArtifact().compareTo(node.getArtifact()) != 0) {
290                     files.addAll(this.dependencies(child, scps));
291                 }
292             }
293         }
294         return files;
295     }

297     /**
298      * Default scopes.
299      * @return List of scopes.
300      */
301     private List<String> scope() {
302         final List<String> scps;
303         if (this.eclipseAether()) {
304             scps = Arrays.asList(
305                 org.eclipse.aether.util.artifact.JavaScopes.COMPILE,
306                 org.eclipse.aether.util.artifact.JavaScopes.PROVIDED,
307                 org.eclipse.aether.util.artifact.JavaScopes.RUNTIME,
308                 org.eclipse.aether.util.artifact.JavaScopes.SYSTEM
309             );
310         } else {
311             scps = Arrays.asList(
312                 JavaScopes.COMPILE,
313                 JavaScopes.PROVIDED,
314                 JavaScopes.RUNTIME,
315                 JavaScopes.SYSTEM
316             );
317         }
318         return scps;
319     }

342         return new StringBuilder()

349     /**
350      * Get locations of all source roots (with aspects in source form).
351      * @return Directories separated
352      */
353     @Cacheable(forever = true)
354     private String sourceroots() {
355         String path;
356         if (this.aspectDirectories == null
357             || this.aspectDirectories.length == 0) {
358             path = Files.createTempDir().getAbsolutePath();
359         } else {
360             for (File dir : this.aspectDirectories) {
361                 if (!dir.exists()) {
362                     throw new IllegalStateException(
363                         String.format("source directory %s is absent", dir)
364                     );
365                 }
366             }
367             path = StringUtils.join(this.aspectDirectories, AjcMojo.SEP);
368         }
369         return path;
370     }

372     /**
373      * Find all files in the directory.
374      * @param dir The directory
375      * @return List of them
376      */
377     private static Collection<File> files(final File dir) {
378         final Collection<File> files = new LinkedList<File>();
379         final Collection<File> all = FileUtils.listFiles(
380             dir, TrueFileFilter.INSTANCE, TrueFileFilter.INSTANCE
381         );
382         for (File file : all) {
383             if (file.isFile()) {
384                 files.add(file);
385             }
386         }
387         return files;
388     }

399     /**
400      * Message holder.
401      */
402     private static final class MsgHolder implements IMessageHolder {
403         /**
404          * All messages seen so far.
405          */
406         private final transient Collection<IMessage> messages =
407             new CopyOnWriteArrayList<IMessage>();
408         /**
409          * {@inheritDoc}
410          */
411         @Override
412         public boolean hasAnyMessage(final IMessage.Kind kind,
413             final boolean greater) {
414             boolean has = false;
415             for (IMessage msg : this.messages) {
416                 has = msg.getKind().equals(kind) || greater
417                     && IMessage.Kind.COMPARATOR
418                     .compare(msg.getKind(), kind) > 0;
419                 if (has) {
420                     break;
421                 }
422             }
423             return has;
424         }
425         /**
426          * {@inheritDoc}
427          */
428         @Override
429         public int numMessages(final IMessage.Kind kind,
430             final boolean greater) {
431             int num = 0;
432             for (IMessage msg : this.messages) {
433                 final boolean has = msg.getKind().equals(kind) || greater
434                     && IMessage.Kind.COMPARATOR
435                     .compare(msg.getKind(), kind) > 0;
436                 if (has) {
437                     ++num;
438                 }
439             }
440             return num;
441         }
442         /**
443          * {@inheritDoc}
444          */
445         @Override
446         public IMessage[] getMessages(final IMessage.Kind kind,
447             final boolean greater) {
448             throw new UnsupportedOperationException();
449         }
450         /**
451          * {@inheritDoc}
452          */
453         @Override
454         public List<IMessage> getUnmodifiableListView() {
455             throw new UnsupportedOperationException();
456         }
457         /**
458          * {@inheritDoc}
459          */
460         @Override
461         public void clearMessages() {
462             throw new UnsupportedOperationException();
463         }
464         /**
465          * {@inheritDoc}
466          */
467         @Override
468         public boolean handleMessage(final IMessage msg) {
469             if (msg.getKind().equals(IMessage.ERROR)
470                 || msg.getKind().equals(IMessage.FAIL)
471                 || msg.getKind().equals(IMessage.ABORT)) {
472                 Logger.error(AjcMojo.class, msg.getMessage());
473             } else if (msg.getKind().equals(IMessage.WARNING)) {
474                 Logger.warn(AjcMojo.class, msg.getMessage());
475             } else if (msg.getKind().equals(IMessage.WEAVEINFO)) {
476                 Logger.debug(AjcMojo.class, msg.getMessage());
477             } else if (msg.getKind().equals(IMessage.INFO)) {
478                 Logger.debug(AjcMojo.class, msg.getMessage());
479             } else {
480                 Logger.debug(AjcMojo.class, msg.getMessage());
481             }
482             this.messages.add(msg);
483             return true;
484         }
485         /**
486          * {@inheritDoc}
487          */
488         @Override
489         public boolean isIgnoring(final IMessage.Kind kind) {
490             return false;
491         }
492         /**
493          * {@inheritDoc}
494          */
495         @Override
496         public void dontIgnore(final IMessage.Kind kind) {
497             assert kind != null;
498         }
499         /**
500          * {@inheritDoc}
501          */
502         @Override
503         public void ignore(final IMessage.Kind kind) {
504             assert kind != null;
505         }
506     }
