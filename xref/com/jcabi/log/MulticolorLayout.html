43  /**
44   * Multi-color layout for LOG4J.
45   *
46   * <p>Use it in your LOG4J configuration:
47   *
48   * <pre> log4j.rootLogger=INFO, CONSOLE
49   * log4j.appender.CONSOLE=org.apache.log4j.ConsoleAppender
50   * log4j.appender.CONSOLE.layout=com.jcabi.log.MulticolorLayout
51   * log4j.appender.CONSOLE.layout.ConversionPattern=[%color{%-5p}] %c: %m%n</pre>
52   *
53   * <p>The part of the message wrapped with {@code %color{...}}
54   * will change its color according to the logging level of the event. Without
55   * this highlighting the behavior of the layout is identical to
56   * {@link PatternLayout}. You can use {@code %color-red{...}} if you
57   * want to use specifically red color for the wrapped piece of text. Supported
58   * colors are: {@code red}, {@code blue}, {@code yellow}, {@code cyan},
59   * {@code black}, and {@code white}.
60   *
61   * <p>Besides that you can specify any ANSI color you like with
62   * {@code %color-<attr>;<bg>;<fg>{...}}, where
63   * {@code <attr>} is a binary mask of attributes,
64   * {@code <bg>} is a background color, and
65   * {@code <fg>} is a foreground color. Read more about
66   * <a href="http://en.wikipedia.org/wiki/ANSI_escape_code">ANSI escape code</a>.
67   *
68   * <p>Maven dependency for this class is
69   * (see <a href="http://www.jcabi.com/jcabi-log/multicolor.html">How
70   * to use with Maven</a> instructions):
71   *
72   * <pre>&lt;dependency&gt;
73   *  &lt;groupId&gt;com.jcabi&lt;/groupId&gt;
74   *  &lt;artifactId&gt;jcabi-log&lt;/artifactId&gt;
75   * &lt;/dependency&gt;</pre>
76   *
77   * @author Yegor Bugayenko (yegor@tpc2.com)
78   * @version $Id$
79   * @since 0.1.10
80   * @see <a href="http://en.wikipedia.org/wiki/ANSI_escape_code">ANSI escape code</a>
81   * @see <a href="http://logging.apache.org/log4j/1.2/apidocs/org/apache/log4j/PatternLayout.html">PatternLayout from LOG4J</a>
82   * @see <a href="http://www.jcabi.com/jcabi-log/multicolor.html">How to use with Maven</a>
83   */
84  @Immutable
85  @ToString
86  @EqualsAndHashCode(callSuper = false)
87  @SuppressWarnings("PMD.NonStaticInitializer")
88  public final class MulticolorLayout extends EnhancedPatternLayout {
89  
90      /**
91       * Control sequence indicator.
92       */
93      private static final String CSI = "\u001b[";
94  
95      /**
96       * Colors with names.
97       */
98      private static final ConcurrentMap<String, String> COLORS =
99          new ConcurrentHashMap<String, String>() {
100             private static final long serialVersionUID = 0x7526EF78EEDFE465L;
101             {
102                 this.put("black", "30");
103                 this.put("blue", "34");
104                 this.put("cyan", "36");
105                 this.put("green", "32");
106                 this.put("magenta", "35");
107                 this.put("red", "31");
108                 this.put("yellow", "33");
109                 this.put("white", "37");
110             }
111         };
112 
113     /**
114      * Colors of levels.
115      */
116     private static final ConcurrentMap<String, String> LEVELS =
117         new ConcurrentHashMap<String, String>() {
118             private static final long serialVersionUID = 0x7526FF78EEDFC465L;
119             {
120                 this.put(Level.TRACE.toString(), "2;33");
121                 this.put(Level.DEBUG.toString(), "2;37");
122                 this.put(Level.INFO.toString(), "0;37");
123                 this.put(Level.WARN.toString(), "0;33");
124                 this.put(Level.ERROR.toString(), "0;31");
125                 this.put(Level.FATAL.toString(), "0;35");
126             }
127         };
128 
129     /**
130      * Regular expression for all matches.
131      */
132     private static final Pattern METAS = Pattern.compile(
133         "%color(?:-([a-z]+|[0-9]{1,3};[0-9]{1,3};[0-9]{1,3}))?\\{(.*?)\\}"
134     );
135 
136     /**
137      * {@inheritDoc}
138      */
139     @Override
140     public void setConversionPattern(final String pattern) {
141         final Matcher matcher = MulticolorLayout.METAS.matcher(pattern);
142         final StringBuffer buf = new StringBuffer();
143         while (matcher.find()) {
144             matcher.appendReplacement(buf, "");
145             buf.append(MulticolorLayout.CSI)
146                 .append(this.ansi(matcher.group(1)))
147                 .append('m')
148                 .append(matcher.group(2))
149                 .append(MulticolorLayout.CSI)
150                 .append('m');
151         }
152         matcher.appendTail(buf);
153         super.setConversionPattern(buf.toString());
154     }
155 
156     /**
157      * {@inheritDoc}
158      */
159     @Override
160     public String format(final LoggingEvent event) {
161         return super.format(event).replace(
162             String.format("%s?m", MulticolorLayout.CSI),
163             String.format(
164                 "%s%sm",
165                 MulticolorLayout.CSI,
166                 MulticolorLayout.LEVELS.get(event.getLevel().toString())
167             )
168         );
169     }
170 
171     /**
172      * Convert our text to ANSI color.
173      * @param meta Meta text
174      * @return ANSI color
175      */
176     private String ansi(final String meta) {
177         String ansi;
178         if (meta == null) {
179             ansi = "?";
180         } else if (meta.matches("[a-z]+")) {
181             ansi = MulticolorLayout.COLORS.get(meta);
182             if (ansi == null) {
183                 throw new IllegalArgumentException(
184                     String.format("unknown color '%s'", meta)
185                 );
186             }
187         } else {
188             ansi = meta;
189         }
190         return ansi;
191     }
