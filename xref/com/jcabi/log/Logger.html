41  /**
42   * Universal logger, and adapter between your app and SLF4J API.
43   *
44   * <p>Instead of relying
45   * on some logging engine you can use this class, which transforms all
46   * messages to SLF4J. This approach gives you a perfect decoupling of business
47   * logic and logging mechanism. All methods in the class are called
48   * statically, without the necessity to instantiate the class.
49   *
50   * <p>Use it like this in any class, and in any package:
51   *
52   * <pre> package com.example.XXX;
53   * import com.jcabi.log.Logger;
54   * public class MyClass {
55   *   public void foo(Integer num) {
56   *     Logger.info(this, "foo(%d) just called", num);
57   *   }
58   * }</pre>
59   *
60   * <p>Or statically (pay attention to {@code MyClass.class}):
61   *
62   * <pre> public class MyClass {
63   *   public static void foo(Integer num) {
64   *     Logger.info(MyClass.class, "foo(%d) just called", num);
65   *   }
66   * }</pre>
67   *
68   * <p>Exact binding between SLF4J and logging facility has to be
69   * specified in {@code pom.xml} of your project (or in classpath directly).
70   *
71   * <p>For performance reasons in most cases before sending a
72   * {@code TRACE} or {@code DEBUG} log message you may check whether this
73   * logging level is enabled in the project, e.g.:
74   *
75   * <pre> //...
76   * if (Logger.isTraceEnabled(this)) {
77   *   Logger.trace(this, "#foo() called");
78   * }
79   * //...</pre>
80   *
81   * <p>There is only one reason to do so - if you want to save time spent on
82   * preparing of the arguments. By default, such a call is made inside every
83   * method of {@link Logger} class.
84   *
85   * @author Yegor Bugayenko (yegor@tpc2.com)
86   * @version $Id$
87   * @since 0.1
88   */

102     /**
103      * Format one string.
104      * @param fmt The format
105      * @param args List of arbitrary arguments
106      * @return Formatted string
107      */
108     public static String format(final String fmt, final Object... args) {
109         String result;
110         if (args.length == 0) {
111             result = fmt;
112         } else {
113             final PreFormatter pre = new PreFormatter(fmt, args);
114             result = String.format(pre.getFormat(), pre.getArguments());
115         }
116         return result;
117     }

282     /**
283      * Log one line using the logging level specified.
284      * @param level The level of logging
285      * @param source The source of the logging operation
286      * @param msg The text message to be logged
287      * @param args Optional arguments for string formatting
288      * @since 0.8
289      * @checkstyle ParameterNumber (4 lines)
290      */
291     public static void log(final Level level, final Object source,
292         final String msg, final Object... args) {
293         if (level.equals(Level.SEVERE)) {
294             Logger.error(source, msg, args);
295         } else if (level.equals(Level.WARNING)) {
296             Logger.warn(source, msg, args);
297         } else if (level.equals(Level.INFO) || level.equals(Level.CONFIG)) {
298             Logger.info(source, msg, args);
299         } else if (level.equals(Level.FINE) || level.equals(Level.ALL)) {
300             Logger.debug(source, msg, args);
301         } else if (level.equals(Level.FINER) || level.equals(Level.FINEST)) {
302             Logger.trace(source, msg, args);
303         }
304     }

306     /**
307      * Returns an {@link OutputStream}, which converts all incoming data
308      * into logging lines (separated by {@code \x0A} in UTF-8).
309      * @param level The level of logging
310      * @param source The source of the logging operation
311      * @return Output stream directly pointed to the logging facility
312      * @since 0.8
313      * @see <a href="http://stackoverflow.com/questions/17258325">some discussion</a>
314      */
315     public static OutputStream stream(final Level level, final Object source) {
316         return new OutputStream() {
317             private final transient ByteArrayOutputStream buffer =
318                 new ByteArrayOutputStream();
319             @Override
320             public void write(final int data) throws IOException {
321                 if (data == '\n') {
322                     Logger.log(level, source, this.buffer.toString("UTF-8"));
323                     this.buffer.reset();
324                 } else {
325                     this.buffer.write(data);
326                 }
327             }
328         };
329     }

331     /**
332      * Get the instance of the logger for this particular caller.
333      * @param source Source of the logging operation
334      * @return The instance of {@link Logger} class
335      */
336     private static org.slf4j.Logger logger(final Object source) {
337         org.slf4j.Logger logger;
338         if (source instanceof Class) {
339             logger = LoggerFactory.getLogger((Class) source);
340         } else {
341             logger = LoggerFactory.getLogger(source.getClass());
342         }
343         return logger;
344     }
