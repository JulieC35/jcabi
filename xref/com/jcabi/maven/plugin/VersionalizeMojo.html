58  /**
59   * Versionalize Java packages.
60   *
61   * @author Yegor Bugayenko (yegor@tpc2.com)
62   * @version $Id$
63   * @since 0.7.16
64   */
65  @MojoGoal("versionalize")
66  @MojoPhase("prepare-package")
67  @ToString
68  @EqualsAndHashCode(callSuper = false)
69  @Loggable(Loggable.DEBUG)
70  @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
71  public final class VersionalizeMojo extends AbstractMojo {
72  
73      /**
74       * Maven project.
75       */
76      @MojoParameter(
77          expression = "${project}",
78          required = true,
79          readonly = true,
80          description = "Maven project"
81      )
82      private transient MavenProject project;
83  
84      /**
85       * Build number.
86       */
87      @MojoParameter(
88          expression = "${buildNumber}",
89          required = false,
90          readonly = false,
91          description = "Build number"
92      )
93      private transient String buildNumber;
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      @Loggable(value = Loggable.DEBUG, limit = 1, unit = TimeUnit.MINUTES)
100     public void execute() throws MojoFailureException {
101         StaticLoggerBinder.getSingleton().setMavenLog(this.getLog());
102         final File src = new File(this.project.getBuild().getSourceDirectory());
103         if (!src.exists()) {
104             Logger.info(this, "source directory '%s' is absent", src);
105             return;
106         }
107         final File dest =
108             new File(this.project.getBuild().getOutputDirectory());
109         dest.mkdirs();
110         Logger.info(this, "Versionalizing %s directory", dest);
111         try {
112             this.versionalize(src, dest);
113         } catch (IOException ex) {
114             throw new MojoFailureException("failed to versionalize", ex);
115         }
116     }
117 
118     /**
119      * Create and return a text of the version file.
120      * @param dir The destination directory
121      * @return The text
122      */
123     private String text(@NotNull final File dir) {
124         final StringBuilder text = new StringBuilder()
125             .append(String.format("Build Number: %s\n", this.buildNumber))
126             .append(
127                 String.format(
128                     "Project Version: %s\n",
129                     this.project.getVersion()
130                 )
131             )
132             .append(
133                 String.format(
134                     "Build Date: %s\n\n",
135                     DateFormatUtils.ISO_DATETIME_FORMAT.format(new Date())
136                 )
137             );
138         for (String name : VersionalizeMojo.files(dir, "*")) {
139             final File file = new File(dir, name);
140             if (file.isFile()) {
141                 text.append(name)
142                     .append(": ")
143                     .append(file.length())
144                     .append("\n");
145             }
146         }
147         return text.toString();
148     }
149 
150     /**
151      * Versionalize packages from source to dest.
152      * @param src Source directory
153      * @param dest Destination
154      * @throws IOException If some IO problem
155      */
156     private void versionalize(@NotNull final File src, @NotNull final File dest)
157         throws IOException {
158         final Collection<File> dirs = FileUtils.listFilesAndDirs(
159             src,
160             new NotFileFilter(TrueFileFilter.INSTANCE),
161             DirectoryFileFilter.DIRECTORY
162         );
163         final String name = String.format(
164             "%s-%s-%s.txt",
165             VersionalizeMojo.cleanup(this.project.getGroupId()),
166             VersionalizeMojo.cleanup(this.project.getArtifactId()),
167             VersionalizeMojo.cleanup(this.project.getPackaging())
168         );
169         for (File dir : dirs) {
170             if (VersionalizeMojo.files(dir, "*.java").isEmpty()) {
171                 continue;
172             }
173             final File ddir = new File(
174                 dest,
175                 dir.getCanonicalPath().substring(
176                     src.getCanonicalPath().length() + 1
177                 )
178             );
179             final File version = new File(ddir, name);
180             version.getParentFile().mkdirs();
181             FileUtils.write(version, this.text(ddir));
182             Logger.info(this, "File %s added", version);
183         }
184     }
185 
186     /**
187      * Clean the text.
188      * @param text The text
189      * @return Clean version of it
190      */
191     private static String cleanup(final String text) {
192         return text.replaceAll("[^_a-z0-9\\-]", "-");
193     }
194 
195     /**
196      * All Java files in the directory.
197      * @param dir The directory
198      * @param mask Mask to use
199      * @return List of Java file names
200      */
201     private static Collection<String> files(final File dir, final String mask) {
202         final FileFilter filter = new WildcardFileFilter(mask);
203         final File[] files = dir.listFiles(filter);
204         final Collection<String> names = new ArrayList<String>(files.length);
205         for (File file : files) {
206             names.add(file.getName());
207         }
208         return names;
209     }
