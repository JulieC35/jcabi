55  /**
56   * Running instances of MySQL.
57   *
58   * <p>The class is thread-safe.

60   * @author Yegor Bugayenko (yegor@tpc2.com)

66  @ToString
67  @EqualsAndHashCode(of = "processes")
68  @Loggable(Loggable.INFO)
69  @SuppressWarnings("PMD.DoNotUseThreads")
70  final class Instances {

90      private final transient ConcurrentMap<Integer, Process> processes =

93      /**
94       * Start a new one at this port.
95       * @param port The port to start at
96       * @param dist Path to MySQL distribution
97       * @param target Where to keep temp data
98       * @throws IOException If fails to start
99       */
100     public void start(final int port, @NotNull final File dist,
101         @NotNull final File target) throws IOException {
102         synchronized (this.processes) {
103             if (this.processes.containsKey(port)) {
104                 throw new IllegalArgumentException(
105                     String.format("port %d is already busy", port)
106                 );
107             }
108             final Process proc = this.process(port, dist, target);
109             this.processes.put(port, proc);
110             Runtime.getRuntime().addShutdownHook(
111                 new Thread(
112                     new Runnable() {
113                         @Override
114                         public void run() {
115                             Instances.this.stop(port);
116                         }
117                     }
118                 )
119             );
120         }
121     }
122 
123     /**
124      * Stop a running one at this port.
125      * @param port The port to stop at
126      */
127     public void stop(final int port) {
128         synchronized (this.processes) {
129             final Process proc = this.processes.get(port);
130             if (proc != null) {
131                 proc.destroy();
132                 this.processes.remove(proc);
133             }
134         }
135     }

137     /**
138      * Start a new process.
139      * @param port The port to start at
140      * @param dist Path to MySQL distribution
141      * @param target Where to keep temp data
142      * @return Process started
143      * @throws IOException If fails to start
144      */
145     private Process process(final int port, final File dist, final File target)
146         throws IOException {
147         if (target.exists()) {
148             FileUtils.deleteDirectory(target);
149             Logger.info(this, "deleted %s directory", target);
150         }
151         if (target.mkdirs()) {
152             Logger.info(this, "created %s directory", target);
153         }
154         new File(target, "temp").mkdirs();
155         final File socket = new File(target, "mysql.sock");
156         final ProcessBuilder builder = this.builder(
157             dist,
158             "bin/mysqld",
159             "--general_log",
160             "--console",
161             "--innodb_buffer_pool_size=64M",
162             "--innodb_log_file_size=64M",
163             "--log_warnings",
164             "--innodb_use_native_aio=0",
165             String.format("--binlog-ignore-db=%s", Instances.DBNAME),
166             String.format("--basedir=%s", dist),
167             String.format("--lc-messages-dir=%s", new File(dist, "share")),
168             String.format("--datadir=%s", this.data(dist, target)),
169             String.format("--tmpdir=%s", new File(target, "temp")),
170             String.format("--socket=%s", socket),
171             String.format("--pid-file=%s", new File(target, "mysql.pid")),
172             String.format("--port=%d", port)
173         ).redirectErrorStream(true);
174         builder.environment().put("MYSQL_HOME", dist.getAbsolutePath());
175         final Process proc = builder.start();
176         final Thread thread = new Thread(
177             new VerboseRunnable(
178                 new Callable<Void>() {
179                     @Override
180                     public Void call() throws Exception {
181                         new VerboseProcess(proc).stdout();
182                         return null;
183                     }
184                 }
185             )
186         );
187         thread.setDaemon(true);
188         thread.start();
189         this.configure(dist, port, this.waitFor(socket, port));
190         return proc;
191     }

193     /**
194      * Prepare and return data directory.
195      * @param dist Path to MySQL distribution
196      * @param target Where to create it
197      * @return Directory created
198      * @throws IOException If fails
199      */
200     private File data(final File dist, final File target) throws IOException {
201         final File dir = new File(target, "data");
202         FileUtils.writeStringToFile(
203             new File(dist, "support-files/my-default.cnf"),
204             "[mysql]\n# no defaults..."
205         );
206         new VerboseProcess(
207             this.builder(
208                 dist,
209                 "scripts/mysql_install_db",
210                 "--no-defaults",
211                 "--force",
212                 "--innodb_use_native_aio=0",
213                 String.format("--datadir=%s", dir)
214             )
215         ).stdout();
216         return dir;
217     }

219     /**
220      * Wait for this file to become available.
221      * @param socket The file to wait for
222      * @param port Port to wait for
223      * @return The same socket, but ready for usage
224      * @throws IOException If fails
225      */
226     private File waitFor(final File socket, final int port) throws IOException {
227         final long start = System.currentTimeMillis();
228         long age = 0;
229         while (true) {
230             if (socket.exists()) {
231                 Logger.info(
232                     this,
233                     "socket %s is available after %[ms]s of waiting",
234                     socket, age
235                 );
236                 break;
237             }
238             if (Instances.isOpen(port)) {
239                 Logger.info(
240                     this,
241                     "port %s is available after %[ms]s of waiting",
242                     port, age
243                 );
244                 break;
245             }
246             try {
247                 TimeUnit.SECONDS.sleep(1);
248             } catch (InterruptedException ex) {
249                 Thread.currentThread().interrupt();
250                 throw new IllegalStateException(ex);
251             }
252             age = System.currentTimeMillis() - start;
253             if (age > TimeUnit.MINUTES.toMillis(1)) {
254                 throw new IOException(
255                     Logger.format(
256                         "socket %s is not available after %[ms]s of waiting",
257                         socket, age
258                     )
259                 );
260             }
261         }
262         return socket;
263     }

265     /**
266      * Configure the running MySQL server.
267      * @param dist Directory with MySQL distribution
268      * @param port The port it's running on
269      * @param socket Socket of it
270      * @throws IOException If fails
271      */
272     private void configure(final File dist, final int port, final File socket)
273         throws IOException {
274         new VerboseProcess(
275             this.builder(
276                 dist,
277                 "bin/mysqladmin",
278                 String.format("--port=%d", port),
279                 String.format("--user=%s", Instances.USER),
280                 String.format("--socket=%s", socket),
281                 "--host=127.0.0.1",
282                 "password",
283                 Instances.PASSWORD
284             )
285         ).stdout();
286         final Process process = this.builder(
287             dist,
288             "bin/mysql",
289             String.format("--port=%d", port),
290             String.format("--user=%s", Instances.USER),
291             String.format("--password=%s", Instances.PASSWORD),
292             String.format("--socket=%s", socket)
293         ).start();
294         final PrintWriter writer = new PrintWriter(
295             new OutputStreamWriter(
296                 process.getOutputStream(), CharEncoding.UTF_8
297             )
298         );
299         writer.print("CREATE DATABASE ");
300         writer.print(Instances.DBNAME);
301         writer.println(";");
302         writer.close();
303         new VerboseProcess(process).stdout();
304     }

306     /**
307      * Make process builder with this commands.
308      * @param dist Distribution directory
309      * @param name Name of the cmd to run
310      * @param cmds Commands
311      * @return Process builder
312      */
313     private ProcessBuilder builder(final File dist, final String name,
314         final String... cmds) {
315         String label = name;
316         final Collection<String> commands = new LinkedList<String>();
317         if (!new File(dist, label).exists()) {
318             label = String.format("%s.exe", name);
319             if (!new File(dist, label).exists()) {
320                 label = String.format("%s.pl", name);
321                 commands.add("perl");
322             }
323         }
324         commands.add(new File(dist, label).getAbsolutePath());
325         commands.addAll(Arrays.asList(cmds));
326         Logger.info(this, "$ %s", StringUtils.join(commands, " "));
327         return new ProcessBuilder()
328             .command(commands.toArray(new String[commands.size()]))
329             .directory(dist);
330     }

332     /**
333      * Port is open.
334      * @param port The port to check
335      * @return TRUE if it's open
336      */
337     private static boolean isOpen(final int port) {
338         boolean open;
339         try {
340             new Socket((String) null, port);
341             open = true;
342         } catch (IOException ex) {
343             open = false;
344         }
345         return open;
346     }
