101     /**
102      * Cacheable can cache just once.
103      * @throws Exception If something goes wrong
104      */
105     @Test
106     public void cachesJustOnceInParallelThreads() throws Exception {
107         final CacheableTest.Foo foo = new CacheableTest.Foo(1);
108         final Thread never = new Thread(
109             new Runnable() {
110                 @Override
111                 public void run() {
112                     foo.never();
113                 }
114             }
115         );
116         never.start();
117         final Set<String> values = new ConcurrentSkipListSet<String>();
118         final int threads = Runtime.getRuntime().availableProcessors() * 2;
119         final CountDownLatch start = new CountDownLatch(1);
120         final CountDownLatch done = new CountDownLatch(threads);
121         final Callable<Boolean> task = new Callable<Boolean>() {
122             @Override
123             public Boolean call() throws Exception {
124                 start.await(1, TimeUnit.SECONDS);
125                 values.add(foo.get().toString());
126                 done.countDown();
127                 return true;
128             }
129         };
130         for (int pos = 0; pos < threads; ++pos) {
131             Executors.newFixedThreadPool(threads).submit(task);
132         }
133         start.countDown();
134         done.await(1, TimeUnit.SECONDS);
135         MatcherAssert.assertThat(values.size(), Matchers.equalTo(1));
136         never.interrupt();
137     }

142     private static final class Foo {
143         /**
144          * Random.
145          */
146         private static final Random RANDOM = new Random();
147         /**
148          * Encapsulated long.
149          */
150         private final transient long number;
151         /**
152          * Public ctor.
153          * @param num Number to encapsulate
154          */
155         public Foo(final long num) {
156             this.number = num;
157         }
158         @Override
159         public int hashCode() {
160             return this.get().hashCode();
161         }
162         @Override
163         public boolean equals(final Object obj) {
164             return obj == this;
165         }
166         @Override
167         @Cacheable(forever = true)
168         @Loggable(Loggable.DEBUG)
169         public String toString() {
170             return Long.toString(this.number);
171         }
172         /**
173          * Download some text.
174          * @return Downloaded text
175          */
176         @Cacheable(lifetime = 1, unit = TimeUnit.SECONDS)
177         @Loggable(Loggable.DEBUG)
178         public CacheableTest.Foo get() {
179             return new CacheableTest.Foo(CacheableTest.Foo.RANDOM.nextLong());
180         }
181         /**
182          * Sleep forever, to abuse caching system.
183          * @return The same object
184          */
185         @Cacheable(lifetime = 1, unit = TimeUnit.SECONDS)
186         @Loggable(Loggable.DEBUG)
187         public CacheableTest.Foo never() {
188             try {
189                 TimeUnit.HOURS.sleep(1);
190             } catch (InterruptedException ex) {
191                 throw new IllegalStateException(ex);
192             }
193             return this;
194         }
195         /**
196          * Flush it.
197          */
198         @Cacheable.FlushBefore
199         public void flush() {
200             // nothing to do
201         }
202         /**
203          * Download some text.
204          * @return Downloaded text
205          */
206         @Cacheable(lifetime = 1, unit = TimeUnit.SECONDS)
207         public static String staticGet() {
208             return Long.toString(CacheableTest.Foo.RANDOM.nextLong());
209         }
210         /**
211          * Flush it.
212          */
213         @Cacheable.FlushBefore
214         public static void staticFlush() {
215             // nothing to do
216         }
217     }
