39  /**
40   * Makes a method loggable via {@link com.jcabi.log.Logger}.
41   *
42   * <p>For example, this {@code load()} method produces a log line
43   * on every call:
44   *
45   * <pre> &#64;Loggable
46   * String load(String resource) throws IOException {
47   *   return "something";
48   * }</pre>
49   *
50   * <p>You can configure the level of logging:
51   *
52   * <pre> &#64;Loggable(Loggable.DEBUG)
53   * void save(String resource) throws IOException {
54   *   // do something
55   * }</pre>
56   *
57   * <p>Since version 0.7.6, you can specify a maximum execution time limit for
58   * a method. If such a limit is reached a logging message will be issued with
59   * a {@code WARN} priority. It is a very convenient mechanism for profiling
60   * applications in production. Default value of a limit is 1 second.
61   *
62   * <pre> &#64;Loggable(limit = 2)
63   * void save(String resource) throws IOException {
64   *   // do something, potentially slow
65   * }</pre>
66   *
67   * <p>Since version 0.7.14 you can change the time unit for the "limit"
68   * parameter. Default unit of measurement is a second:
69   *
70   * <pre> &#64;Loggable(limit = 200, unit = TimeUnit.MILLISECONDS)
71   * void save(String resource) throws IOException {
72   *   // do something, potentially slow
73   * }</pre>
74   *
75   * <p>Since version 0.7.17 you can ignore certain exception types, and they
76   * won't be logged when thrown. It is very useful when exceptions are used
77   * to control flow (which is not a good practice, but is still used in
78   * some frameworks, for example in JAX-RS):
79   *
80   * <pre> &#64;Loggable(ignore = WebApplicationException.class)
81   * String get() {
82   *   if (not_logged_in()) {
83   *     throw new WebApplicationException(forward_to_login_page());
84   *   }
85   * }</pre>
86   *
87   * <p>Since version 0.8 you can mark some exceptions as "always to be ignored",
88   * using {@link Loggable.Quiet} annotation.
89   *
90   * @author Yegor Bugayenko (yegor@tpc2.com)
91   * @version $Id$
92   * @since 0.7.2
93   * @see com.jcabi.log.Logger
94   * @see <a href="http://www.jcabi.com/jcabi-aspects">http://www.jcabi.com/jcabi-aspects/</a>
95   */
96  @Documented
97  @Retention(RetentionPolicy.RUNTIME)
98  @Target({ ElementType.METHOD, ElementType.CONSTRUCTOR, ElementType.TYPE })
99  @SuppressWarnings("PMD.VariableNamingConventions")
100 public @interface Loggable {
101 
102     /**
103      * TRACE level of logging.
104      */
105     int TRACE = 0;
106 
107     /**
108      * DEBUG level of logging.
109      */
110     int DEBUG = 1;
111 
112     /**
113      * INFO level of logging.
114      */
115     int INFO = 2;
116 
117     /**
118      * WARN level of logging.
119      */
120     int WARN = 3;
121 
122     /**
123      * ERROR level of logging.
124      */
125     int ERROR = 4;
126 
127     /**
128      * Level of logging.
129      */
130     int value() default Loggable.INFO;
131 
132     /**
133      * Maximum amount allowed for this method (a warning will be
134      * issued if it takes longer).
135      * @since 0.7.6
136      */
137     int limit() default 1;
138 
139     /**
140      * Time unit for the limit.
141      * @since 0.7.14
142      */
143     TimeUnit unit() default TimeUnit.MINUTES;
144 
145     /**
146      * Shall we trim long texts in order to make log lines more readable?
147      * @since 0.7.13
148      */
149     boolean trim() default true;
150 
151     /**
152      * Method entry moment should be reported as well (by default only
153      * an exit moment is reported).
154      * @since 0.7.16
155      */
156     boolean prepend() default false;
157 
158     /**
159      * List of exception types, which should not be logged if thrown.
160      *
161      * <p>You can also mark some exception types as "always to be ignored",
162      * using {@link Loggable.Quiet} annotation.
163      *
164      * @since 0.7.17
165      */
166     Class<? extends Throwable>[] ignore() default { };
167 
168     /**
169      * Skip logging of result, replacing it with dots?
170      * @since 0.7.19
171      */
172     boolean skipResult() default false;
173 
174     /**
175      * Skip logging of arguments, replacing them all with dots?
176      * @since 0.7.19
177      */
178     boolean skipArgs() default false;
179 
180     /**
181      * Identifies an exception that is never logged by {@link Loggable} if/when
182      * being thrown out of an annotated method.
183      *
184      * <p>Sometimes exceptions are used as flow control instruments (although
185      * this may be considered as a bad practice in most casts). In such
186      * situations we don't want to flood log console with error messages. One
187      * of the options is to use {@link Loggable#ignore()} attribute to list
188      * all exception types that should be ignored. However, this
189      * {@link Loggable.Quiet} annotation is more convenient when we want to
190      * ignore one specific exception type in all situations.
191      *
192      * @since 0.8
193      */
194     @Documented
195     @Retention(RetentionPolicy.RUNTIME)
196     @Target(ElementType.TYPE)
197     public @interface Quiet {
198     }

200 }
