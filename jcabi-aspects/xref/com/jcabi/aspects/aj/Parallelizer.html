50  /**
51   * Execute method in multiple threads.
52   *
53   * @author Krzysztof Krason (Krzysztof.Krason@gmail.com)
54   * @version $Id$
55   * @since 0.10
56   * @see com.jcabi.aspects.Parallel
57   */
58  @Aspect
59  @Immutable
60  public final class Parallelizer {
61  
62      /**
63       * Execute method in multiple threads.
64       *
65       * <p>This aspect should be used only on void returning methods.
66       *
67       * <p>Try NOT to change the signature of this method, in order to keep
68       * it backward compatible.
69       *
70       * @param point Joint point
71       * @return The result of call
72       * @throws ParallelException If something goes wrong inside
73       * @checkstyle IllegalThrowsCheck (4 lines)
74       */
75      @Around("execution(@com.jcabi.aspects.Parallel * * (..))")
76      public Object wrap(final ProceedingJoinPoint point)
77          throws ParallelException {
78          final int total = ((MethodSignature) point.getSignature())
79              .getMethod().getAnnotation(Parallel.class).threads();
80          final Collection<Callable<Throwable>> callables =
81              new ArrayList<Callable<Throwable>>(total);
82          final CountDownLatch start = new CountDownLatch(1);
83          for (int thread = 0; thread < total; ++thread) {
84              callables.add(this.callable(point, start));
85          }
86          final ExecutorService executor = Executors
87              .newFixedThreadPool(total, new VerboseThreads());
88          final List<Future<Throwable>> futures =
89              new ArrayList<Future<Throwable>>(total);
90          for (Callable<Throwable> callable : callables) {
91              futures.add(executor.submit(callable));
92          }
93          start.countDown();
94          final Collection<Throwable> failures = new LinkedList<Throwable>();
95          for (Future<Throwable> future : futures) {
96              final Throwable exception;
97              try {
98                  exception = future.get();
99                  if (exception != null) {
100                     failures.add(exception);
101                 }
102             } catch (InterruptedException ex) {
103                 failures.add(ex);
104             } catch (ExecutionException ex) {
105                 failures.add(ex);
106             }
107         }
108         if (!failures.isEmpty()) {
109             throw this.exceptions(failures);
110         }
111         return null;
112     }
113 
114     /**
115      * Create parallel exception.
116      * @param failures List of exceptions from threads.
117      * @return Aggregated exceptions.
118      */
119     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
120     private ParallelException exceptions(
121         final Collection<Throwable> failures) {
122         ParallelException current = null;
123         for (Throwable failure : failures) {
124             current = new ParallelException(failure, current);
125         }
126         return current;
127     }
128 
129     /**
130      * Create callable that executes join point.
131      * @param point Join point to use.
132      * @param start Latch to use.
133      * @return Created callable.
134      */
135     @SuppressWarnings("PMD.AvoidCatchingThrowable")
136     private Callable<Throwable> callable(final ProceedingJoinPoint point,
137         final CountDownLatch start) {
138         return new Callable<Throwable>() {
139             @Override
140             public Throwable call() {
141                 Throwable result = null;
142                 try {
143                     start.await();
144                     point.proceed();
145                     // @checkstyle IllegalCatchCheck (1 line)
146                 } catch (Throwable ex) {
147                     result = ex;
148                 }
149                 return result;
150             }
151         };
152     }
153 
154     /**
155      * Exception that encapsulates all exceptions thrown from threads.
156      */
157     private static final class ParallelException extends Exception {
158         /**
159          * Serialization marker.
160          */
161         private static final long serialVersionUID = 0x8743ef363febc422L;
162         /**
163          * Next parallel exception.
164          */
165         private final transient ParallelException next;
166         /**
167          * Constructor.
168          * @param cause Cause of the current exception.
169          * @param nxt Following exception.
170          */
171         protected ParallelException(final Throwable cause,
172             final ParallelException nxt) {
173             super(cause);
174             this.next = nxt;
175         }
176         /**
177          * Get next parallel exception.
178          * @return Next exception.
179          */
180         public ParallelException getNext() {
181             return this.next;
182         }
183     }
