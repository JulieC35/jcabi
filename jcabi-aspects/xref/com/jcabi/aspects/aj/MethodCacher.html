52  /**
53   * Cache method results.
54   *
55   * <p>It is an AspectJ aspect and you are not supposed to use it directly. It
56   * is instantiated by AspectJ runtime framework when your code is annotated
57   * with {@link Cacheable} annotation.
58   *
59   * <p>The class is thread-safe.
60   *
61   * @author Yegor Bugayenko (yegor@tpc2.com)
62   * @version $Id$
63   * @since 0.8
64   */
65  @Aspect
66  @SuppressWarnings({ "PMD.DoNotUseThreads", "PMD.TooManyMethods" })
67  public final class MethodCacher {
68  
69      /**
70       * Calling tunnels.
71       * @checkstyle LineLength (2 lines)
72       */
73      private final transient ConcurrentMap<MethodCacher.Key, MethodCacher.Tunnel> tunnels =
74          new ConcurrentHashMap<MethodCacher.Key, MethodCacher.Tunnel>(0);
75  
76      /**
77       * Service that cleans cache.
78       */
79      private final transient ScheduledExecutorService cleaner =
80          Executors.newSingleThreadScheduledExecutor(
81              new NamedThreads(
82                  "cacheable",
83                  "automated cleaning of expired @Cacheable values"
84              )
85          );
86  
87      /**
88       * Public ctor.
89       */
90      public MethodCacher() {
91          this.cleaner.scheduleWithFixedDelay(
92              new VerboseRunnable(
93                  new Runnable() {
94                      @Override
95                      public void run() {
96                          MethodCacher.this.clean();
97                      }
98                  }
99              ),
100             1, 1, TimeUnit.SECONDS
101         );
102     }

104     /**
105      * Call the method or fetch from cache.
106      *
107      * <p>Try NOT to change the signature of this method, in order to keep
108      * it backward compatible.
109      *
110      * @param point Joint point
111      * @return The result of call
112      * @throws Throwable If something goes wrong inside
113      * @checkstyle IllegalThrows (4 lines)
114      */
115     @Around("execution(* *(..)) && @annotation(com.jcabi.aspects.Cacheable)")
116     public Object cache(final ProceedingJoinPoint point) throws Throwable {
117         final Key key = new MethodCacher.Key(point);
118         Tunnel tunnel;
119         synchronized (this.tunnels) {
120             tunnel = this.tunnels.get(key);
121             if (tunnel == null || tunnel.expired()) {
122                 tunnel = new MethodCacher.Tunnel(point, key);
123                 this.tunnels.put(key, tunnel);
124             }
125         }
126         return tunnel.through();
127     }

174         "execution(* *(..))"
175         + " && @annotation(com.jcabi.aspects.Cacheable.FlushAfter)"

178         this.flush(point, "after the call");
179     }

181     /**
182      * Flush cache.
183      * @param point Joint point
184      * @param when When it happens
185      * @since 0.7.18
186      */
187     private void flush(final JoinPoint point, final String when) {
188         synchronized (this.tunnels) {
189             for (MethodCacher.Key key : this.tunnels.keySet()) {
190                 if (!key.sameTarget(point)) {
191                     continue;
192                 }
193                 final Tunnel removed = this.tunnels.remove(key);
194                 final Method method = MethodSignature.class
195                     .cast(point.getSignature())
196                     .getMethod();
197                 if (Logger.isDebugEnabled(method.getDeclaringClass())) {
198                     Logger.debug(
199                         method.getDeclaringClass(),
200                         "%s: %s:%s removed from cache %s",
201                         Mnemos.toText(method, point.getArgs(), true, false),
202                         key,
203                         removed,
204                         when
205                     );
206                 }
207             }
208         }
209     }
210 
211     /**
212      * Clean cache.
213      */
214     private void clean() {
215         synchronized (this.tunnels) {
216             for (Key key : this.tunnels.keySet()) {
217                 if (this.tunnels.get(key).expired()) {
218                     final Tunnel tunnel = this.tunnels.remove(key);
219                     Logger.debug(
220                         this,
221                         "%s:%s expired in cache",
222                         key,
223                         tunnel
224                     );
225                 }
226             }
227         }
228     }

230     /**
231      * Mutable caching/calling tunnel, it is thread-safe.
232      */
233     private static final class Tunnel {
234         /**
235          * Proceeding join point.
236          */
237         private final transient ProceedingJoinPoint point;
238         /**
239          * Key related to this tunnel.
240          */
241         private final transient Key key;
242         /**
243          * Was it already executed?
244          */
245         private transient boolean executed;
246         /**
247          * When will it expire (moment in time).
248          */
249         private transient long lifetime;
250         /**
251          * Cached value.
252          */
253         private transient Object cached;
254         /**
255          * Public ctor.
256          * @param pnt Joint point
257          * @param akey The key related to it
258          */
259         protected Tunnel(final ProceedingJoinPoint pnt, final Key akey) {
260             this.point = pnt;
261             this.key = akey;
262         }
263         @Override
264         public String toString() {
265             return Mnemos.toText(this.cached, true, false);
266         }
267         /**
268          * Get a result through the tunnel.
269          * @return The result
270          * @throws Throwable If something goes wrong inside
271          * @checkstyle IllegalThrows (5 lines)
272          */
273         @SuppressWarnings("PMD.AvoidSynchronizedAtMethodLevel")
274         public synchronized Object through() throws Throwable {
275             if (!this.executed) {
276                 final long start = System.currentTimeMillis();
277                 this.cached = this.point.proceed();
278                 final Method method = MethodSignature.class
279                     .cast(this.point.getSignature())
280                     .getMethod();
281                 final Cacheable annot = method.getAnnotation(Cacheable.class);
282                 String suffix;
283                 if (annot.forever()) {
284                     this.lifetime = Long.MAX_VALUE;
285                     suffix = "valid forever";
286                 } else if (annot.lifetime() == 0) {
287                     this.lifetime = 0;
288                     suffix = "invalid immediately";
289                 } else {
290                     final long msec = annot.unit().toMillis(annot.lifetime());
291                     this.lifetime = start + msec;
292                     suffix = Logger.format("valid for %[ms]s", msec);
293                 }
294                 final Class<?> type = method.getDeclaringClass();
295                 if (Logger.isDebugEnabled(type)) {
296                     Logger.debug(
297                         type,
298                         "%s: %s cached in %[ms]s, %s",
299                         Mnemos.toText(
300                             method, this.point.getArgs(), true, false
301                         ),
302                         Mnemos.toText(this.cached, true, false),
303                         System.currentTimeMillis() - start,
304                         suffix
305                     );
306                 }
307                 this.executed = true;
308             }
309             return this.key.through(this.cached);
310         }
311         /**
312          * Is it expired already?
313          * @return TRUE if expired
314          */
315         public boolean expired() {
316             return this.executed && this.lifetime < System.currentTimeMillis();
317         }
318     }

320     /**
321      * Key of a callable target.
322      */
323     private static final class Key {
324         /**
325          * When instantiated.
326          */
327         private final transient long start = System.currentTimeMillis();
328         /**
329          * How many times the key was already accessed.
330          */
331         private final transient AtomicInteger accessed = new AtomicInteger();
332         /**
333          * Method.
334          */
335         private final transient Method method;
336         /**
337          * Object callable (or class, if static method).
338          */
339         private final transient Object target;
340         /**
341          * Arguments.
342          */
343         private final transient Object[] arguments;
344         /**
345          * Public ctor.
346          * @param point Joint point
347          */
348         protected Key(final ProceedingJoinPoint point) {
349             this.method = MethodSignature.class
350                 .cast(point.getSignature()).getMethod();
351             this.target = MethodCacher.Key.targetize(point);
352             this.arguments = point.getArgs();
353         }
354         @Override
355         public String toString() {
356             return Mnemos.toText(this.method, this.arguments, true, false);
357         }
358         @Override
359         public int hashCode() {
360             return this.method.hashCode();
361         }
362         @Override
363         public boolean equals(final Object obj) {
364             boolean equals;
365             if (this == obj) {
366                 equals = true;
367             } else if (obj instanceof MethodCacher.Key) {
368                 final MethodCacher.Key key = MethodCacher.Key.class.cast(obj);
369                 equals = key.method.equals(this.method)
370                     && this.target.equals(key.target)
371                     && Arrays.deepEquals(key.arguments, this.arguments);
372             } else {
373                 equals = false;
374             }
375             return equals;
376         }
377         /**
378          * Send a result through, with necessary logging.
379          * @param result The result to send through
380          * @return The same result/object
381          */
382         public Object through(final Object result) {
383             final int hit = this.accessed.getAndIncrement();
384             final Class<?> type = this.method.getDeclaringClass();
385             if (hit > 0 && Logger.isDebugEnabled(type)) {
386                 Logger.debug(
387                     type,
388                     "%s: %s from cache (hit #%d, %[ms]s old)",
389                     this,
390                     Mnemos.toText(result, true, false),
391                     hit,
392                     System.currentTimeMillis() - this.start
393                 );
394             }
395             return result;
396         }
397         /**
398          * Is it related to the same target?
399          * @param point Proceeding point
400          * @return True if the target is the same
401          */
402         public boolean sameTarget(final JoinPoint point) {
403             return MethodCacher.Key.targetize(point).equals(this.target);
404         }
405         /**
406          * Calculate its target.
407          * @param point Proceeding point
408          * @return The target
409          */
410         private static Object targetize(final JoinPoint point) {
411             Object tgt;
412             final Method method = MethodSignature.class
413                 .cast(point.getSignature()).getMethod();
414             if (Modifier.isStatic(method.getModifiers())) {
415                 tgt = method.getDeclaringClass();
416             } else {
417                 tgt = point.getTarget();
418             }
419             return tgt;
420         }
421     }
