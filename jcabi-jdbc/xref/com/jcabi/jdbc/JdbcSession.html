47  /**
48   * Universal JDBC wrapper.
49   *
50   * <p>Execute a simple SQL query over a JDBC data source:
51   *
52   * <pre> String name = new JdbcSession(source)
53   *   .sql("SELECT name FROM foo WHERE id = ?")
54   *   .set(123)
55   *   .select(
56   *     new JdbcSession.Handler&lt;String&gt;() {
57   *       &#64;Override
58   *       public String handle(final ResultSet rset) throws SQLException {
59   *         rset.next();
60   *         return rset.getString(1);
61   *       }
62   *     }
63   *   );</pre>
64   *
65   * <p>There are a number of convenient pre-defined handlers, like
66   * {@link VoidHandler}, {@link NotEmptyHandler}, {@link SingleHandler}, etc.
67   *
68   * <p>Methods {@link #insert(Handler)}, {@link #update(Handler)},
69   * {@link #execute()}, and
70   * {@link #select(Handler)} clean the list of arguments pre-set by
71   * {@link #set(Object)}. The class can be used for a complex transaction, when
72   * it's necessary to perform a number of SQL statements in a group. For
73   * example, the following construct will execute two SQL queries, in a single
74   * transaction and will "commit" at the end (or rollback the entire transaction
75   * in case of any error in between):
76   *
77   * <pre> new JdbcSession(source)
78   *   .autocommit(false)
79   *   .sql("START TRANSACTION")
80   *   .execute()
81   *   .sql("DELETE FROM foo WHERE id = ?")
82   *   .set(444)
83   *   .execute()
84   *   .set(555)
85   *   .execute()
86   *   .commit();</pre>
87   *
88   * <p>The following SQL queries will be sent to the database:
89   *
90   * <pre> START TRANSACTION;
91   * DELETE FROM foo WHERE id = 444;
92   * DELETE FROM foo WHERE id = 555;
93   * COMMIT;</pre>
94   *
95   * <p>{@link #autocommit(boolean)} (with {@code false} as an argument)
96   * can be used when it's necessary to execute
97   * a statement and leave the connection open. For example when shutting down
98   * the database through SQL:
99   *
100  * <pre> new JdbcSession(&#47;* H2 Database data source *&#47;)
101  *   .autocommit(false)
102  *   .sql("SHUTDOWN COMPACT")
103  *   .execute();</pre>
104  *
105  * <p>This class is thread-safe.
106  *
107  * @author Yegor Bugayenko (yegor@tpc2.com)
108  * @version $Id$
109  * @since 0.1.8
110  */

222      * {@link Connection#close()}).

224      */
225     public void commit() throws SQLException {
226         final Connection conn = this.connection.get();
227         if (conn == null) {
228             throw new IllegalStateException(
229                 "connection is not open, can't commit"

233         this.disconnect();

236     /**
237      * Make SQL {@code INSERT} request.
238      *
239      * <p>{@link Handler} will receive a {@link ResultSet} of generated keys.
240      *
241      * <p>JDBC connection is opened and, optionally, closed by this method.
242      *
243      * @param handler The handler or result
244      * @return The result
245      * @param <T> Type of response
246      * @throws SQLException If fails
247      */
248     public <T> T insert(
249         @NotNull(message = "handler can't be NULL") final Handler<T> handler)
250         throws SQLException {
251         return this.run(
252             handler,
253             new Fetcher() {
254                 @Override
255                 public ResultSet fetch(final PreparedStatement stmt)
256                     throws SQLException {
257                     stmt.execute();
258                     return stmt.getGeneratedKeys();
259                 }
260                 @Override
261                 public PreparedStatement statement(final Connection conn)
262                     throws SQLException {
263                     return conn.prepareStatement(
264                         JdbcSession.this.query,
265                         Statement.RETURN_GENERATED_KEYS
266                     );
267                 }
268             }
269         );
270     }

372     /**
373      * Run this handler, and this fetcher.
374      * @param handler The handler or result
375      * @param fetcher Fetcher of result set
376      * @return The result
377      * @param <T> Type of response
378      * @throws SQLException If fails
379      * @checkstyle ExecutableStatementCount (100 lines)
380      */
381     private <T> T run(final Handler<T> handler, final Fetcher fetcher)
382         throws SQLException {
383         if (this.query == null) {
384             throw new IllegalStateException("call #sql() first");
385         }
386         final Connection conn = this.connect();
387         T result;
388         try {
389             conn.setAutoCommit(false);
390             final PreparedStatement stmt = fetcher.statement(conn);
391             try {
392                 this.parametrize(stmt);
393                 final ResultSet rset = fetcher.fetch(stmt);
394                 // @checkstyle NestedTryDepth (5 lines)
395                 try {
396                     result = handler.handle(rset);
397                 } finally {
398                     if (rset != null) {
399                         rset.close();
400                     }
401                 }
402             } finally {
403                 stmt.close();
404             }
405         } catch (SQLException ex) {
406             if (!this.auto) {
407                 conn.rollback();
408                 this.disconnect();
409             }
410             throw new SQLException(ex);
411         } finally {
412             if (this.auto) {
413                 this.commit();
414             }
415             this.args.clear();
416         }
417         return result;
418     }

420     /**
421      * Open connection and cache it locally in the class.
422      * @return Connection to use
423      * @throws SQLException If fails
424      */
425     private Connection connect() throws SQLException {
426         synchronized (this.args) {
427             if (this.connection.get() == null) {
428                 this.connection.set(this.source.getConnection());
429             }
430             return this.connection.get();
431         }
432     }

434     /**
435      * Close connection if it's open (runtime exception otherwise).
436      * @throws SQLException If fails to do the SQL operation
437      */
438     private void disconnect() throws SQLException {
439         final Connection conn = this.connection.getAndSet(null);
440         if (conn == null) {
441             throw new IllegalStateException(
442                 "connection is not open, can't close"
443             );
444         }
445         conn.close();
446     }

448     /**
449      * Add params to the statement.
450      * @param stmt The statement to parametrize
451      * @throws SQLException If some problem
452      */
453     private void parametrize(final PreparedStatement stmt) throws SQLException {
454         int pos = 1;
455         for (Object arg : this.args) {
456             if (arg == null) {
457                 stmt.setString(pos, null);
458             } else if (arg instanceof Long) {
459                 stmt.setLong(pos, Long.class.cast(arg));
460             } else if (arg instanceof Boolean) {
461                 stmt.setBoolean(pos, Boolean.class.cast(arg));
462             } else if (arg instanceof Date) {
463                 stmt.setDate(pos, Date.class.cast(arg));
464             } else if (arg instanceof Integer) {
465                 stmt.setInt(pos, Integer.class.cast(arg));
466             } else if (arg instanceof Utc) {
467                 Utc.class.cast(arg).setTimestamp(stmt, pos);
468             } else {
469                 stmt.setString(pos, arg.toString());
470             }
471             ++pos;
472         }
473     }
